Database name: cinema_system

Employees table:

CREATE TABLE employees (
    id INT(11) NOT NULL AUTO_INCREMENT,
    employeeID VARCHAR(255) DEFAULT NULL,
    first_name VARCHAR(255) DEFAULT NULL,
    last_name VARCHAR(255) DEFAULT NULL,
    email VARCHAR(255) DEFAULT NULL,
    password VARCHAR(255) DEFAULT NULL,
    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;



Movies table:

CREATE TABLE movies (
    id INT(11) NOT NULL AUTO_INCREMENT,
    cinema_room VARCHAR(255) NOT NULL,
    movie_name VARCHAR(255) NOT NULL,
    genre VARCHAR(255) NOT NULL,
    duration VARCHAR(255) NOT NULL,
    price INT(11) NOT NULL,
    mtrbc_rating VARCHAR(255) NOT NULL,
    screening_1 VARCHAR(255) NOT NULL,
    screening_2 VARCHAR(255) NOT NULL,
    screening_3 VARCHAR(255) NOT NULL,
    screening_4 VARCHAR(255) NOT NULL,
    PRIMARY KEY (id),
    UNIQUE KEY `movie_name` (`movie_name`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

Reserved tickets table: (Foreign key sa movies     

CREATE TABLE `reserved_tickets` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `movie_name` varchar(255) NOT NULL,
  `schedule` varchar(255) NOT NULL DEFAULT 'empty',
  `date` varchar(255) NOT NULL DEFAULT 'empty',
  `seat` varchar(255) NOT NULL DEFAULT 'empty',
  `occupant` varchar(255) NOT NULL DEFAULT 'regular',
  `total_price` int(11) NULL,
  PRIMARY KEY (`id`),
  CONSTRAINT `fk_movie_link` 
    FOREIGN KEY (`movie_name`) 
    REFERENCES `movies` (`movie_name`) 
    ON DELETE CASCADE 
    ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;



Admin table:

CREATE TABLE admin (
  id int(11) PRIMARY KEY NOT NULL AUTO_INCREMENT,
  employeeID varchar(255) NOT NULL,
  first_name varchar(255) NOT NULL,
  last_name varchar(255) NOT NULL,
  email varchar(255) NOT NULL,
  password varchar(255) NOT NULL
);
